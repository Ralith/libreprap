project(libreprap)
cmake_minimum_required(VERSION 2.6)

add_definitions(-Wall -Wextra)

add_library(reprap SHARED
  reprap.c
  serial.c
  common.c
  fived.c
  tonokip.c
  comms.c
  util.c
  )

find_package(PkgConfig)
pkg_check_modules(USB libusb-1.0)
if(USB_FOUND)
  add_definitions(-DUSB)
  include_directories(${USB_INCLUDE_DIRS})
  message(STATUS "Building with USB support.")
else(USB_FOUND)
  message(STATUS "WARNING: NOT buiding with USB support.")
endif(USB_FOUND)

install(TARGETS reprap DESTINATION lib)
install(FILES reprap.h comms.h util.h ports.h DESTINATION include/reprap)
